#include <stdio.h>
#include <omp.h>

int main() {
    int n, wind_sz, i, j;

    printf("Enter array size: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Window size: ");
    scanf("%d", &wind_sz);


    if (wind_sz > n || wind_sz <= 0) {
        printf("Invalid window size.\n");
        return 1;
    }


    int num_windows = n - wind_sz + 1;
    int sw[num_windows];

    #pragma omp parallel for private(j)
    for (i = 0; i < num_windows; i++) {
        int current_sum = 0;

        for (j = 0; j < wind_sz; j++) {
            current_sum += arr[i + j];
        }

        sw[i] = current_sum;
    }

    printf("\nSliding Window Sums:\n");
    for (i = 0; i < num_windows; i++) {
        printf("%d ", sw[i]);
    }
    printf("\n");

    return 0;
}
