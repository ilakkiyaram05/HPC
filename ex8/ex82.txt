#include <stdio.h>
#include <omp.h>

int main() {
    long n;
    int i, j;

    printf("Enter matrix size n: ");
    if (scanf("%ld", &n) != 1) return 1;
    int arr[n][n], hori[n][n];

    printf("Enter matrix elements:\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    #pragma omp parallel for private(j)
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            hori[i][j] = arr[i][n - 1 - j];
        }
    }

    printf("Horizontal Mirror Matrix:\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            printf("%d ", hori[i][j]);
        }
        printf("\n");
    }

    return 0;
}
