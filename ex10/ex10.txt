#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <omp.h>

#define MAX_QUEUE 100
#define MAX_LINE 1024

// Queue structure
char *queue[MAX_QUEUE];
int front = 0, rear = 0, count = 0;

// Synchronization
omp_lock_t lock;
int producers_done = 0;

// Enqueue function
void enqueue(char *line) {
    while (1) {
        omp_set_lock(&lock);
        if (count < MAX_QUEUE) {
            queue[rear] = line;
            rear = (rear + 1) % MAX_QUEUE;
            count++;
            omp_unset_lock(&lock);
            break;
        }
        omp_unset_lock(&lock);
    }
}

// Dequeue function
char* dequeue() {
    char *line = NULL;

    while (1) {
        omp_set_lock(&lock);
        if (count > 0) {
            line = queue[front];
            front = (front + 1) % MAX_QUEUE;
            count--;
            omp_unset_lock(&lock);
            return line;
        }
        else if (producers_done) {
            omp_unset_lock(&lock);
            return NULL;
        }
        omp_unset_lock(&lock);
    }
}

// Tokenize and print
void tokenize_and_print(char *line, int tid) {
    char *token = strtok(line, " \t\n");
    while (token != NULL) {
        printf("Thread %d: %s\n", tid, token);
        token = strtok(NULL, " \t\n");
    }
}

// Producer function
void producer(char *filename) {
    FILE *fp = fopen(filename, "r");
    if (!fp) {
        printf("Error opening file %s\n", filename);
        return;
    }

    char buffer[MAX_LINE];

    while (fgets(buffer, MAX_LINE, fp)) {
        char *line = strdup(buffer); // copy line
        enqueue(line);
    }

    fclose(fp);
}

// Consumer function
void consumer(int tid) {
    while (1) {
        char *line = dequeue();
        if (line == NULL)
            break;

        tokenize_and_print(line, tid);
        free(line);
    }
}

int main() {
    int num_producers = 2;
    int num_consumers = 2;

    char *files[] = {"file1.txt", "file2.txt"};

    omp_init_lock(&lock);

    #pragma omp parallel num_threads(num_producers + num_consumers)
    {
        int tid = omp_get_thread_num();

        if (tid < num_producers) {
            producer(files[tid]);

            // Mark producer done
            #pragma omp critical
            {
                producers_done++;
            }
        } else {
            consumer(tid);
        }
    }

    omp_destroy_lock(&lock);

    return 0;
}
