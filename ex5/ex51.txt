#include<stdio.h>
#include<stdlib.h>
#include<pthread.h>
long long global_sum=0;
int thread_count;
long long n;
pthread_mutex_t mutex;
void* thread_sum(void* rank);

int main(int argc,char* argv[]){
   long thread;
   pthread_t* thread_handles;
   if(argc!=3){
      printf("Usage: %s <no.of threads><n>\n",argv[0]);
      return 1;
   }
   thread_count=strtol(argv[1],NULL,10);
   n=strtoll(argv[2],NULL,10);
   thread_handles=malloc(thread_count *sizeof(pthread_t));
   pthread_mutex_init(&mutex,NULL);
   for(thread=0;thread<thread_count;thread++){
      pthread_create(&thread_handles[thread],NULL,thread_sum,(void*)thread);
   }
   for(thread=0;thread<thread_count;thread++){
      pthread_join(thread_handles[thread],NULL);
   }
   printf("The total sum of 1 to %lld is : %lld\n",n,global_sum);
   free(thread_handles);
   return 0;
}
void* thread_sum(void* rank){
   long my_rank=(long)rank;
   long long local_n=n/thread_count;
   long long my_first_i=my_rank*local_n+1;
   long long my_last_i;
   if(my_rank==thread_count-1){
      my_last_i=n;
   }else{
      my_last_i=(my_rank+1)*local_n;
   }
   long long my_sum=0;
   for(long long i=my_first_i;i<=my_last_i;i++){
      my_sum+=i;
   }
   pthread_mutex_lock(&mutex);
   global_sum+=my_sum;
   pthread_mutex_unlock(&mutex);
   return NULL;
}
